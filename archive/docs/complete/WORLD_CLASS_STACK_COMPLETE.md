### TypeScript SDK
```typescript
import { UniversalLegendClient } from '@bobbys-workshop/universal-legend-sdk';

const client = new UniversalLegendClient({
  baseUrl: 'http://localhost:3001',
  apiKey: 'your-api-key',
});

// Get performance stats
const stats = await client.getPerformanceStats();

// Create alert rule
await client.createAlertRule({
  name: 'High Error Rate',
  type: 'threshold',
  metric: 'errors_total',
  operator: 'gt',
  threshold: 100,
});

// Get insights
const insights = await client.getPerformanceInsights();
```
